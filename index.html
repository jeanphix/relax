A basic web application

from flask import Flask
from flask.views import MethodView


app = Flask(__name__)


class HelloView(MethodView):

    def get(self, name):
        return "Hello %s!" % name, 200


app.add_url_rule('/<name>/', HelloView.as_view('hello'),
    _methods=['GET'])
